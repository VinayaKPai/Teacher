SELECT questionbank.Id as 'ID', classes.classNumber as `Class`, subjects.Subject as `Subject`, topics.Name as `Topic`, questiontype.typeName as `Type`, questionbank.question as `Question` FROM questionbank, classes, questiontype, subjects, topics WHERE classes.id = questionbank.classNumber AND subjects.id = questionbank.subjectName AND topics.Id = questionbank.topicName AND questiontype.Id = questionbank.typeName
